00042 /**
00043  * Lookup a commit object from a repository.

00045  * @param commit pointer to the looked up commit
00046  * @param repo the repo to use when locating the commit.
00047  * @param id identity of the commit to locate.  If the object is

00053         return git_object_lookup((git_object **)commit, repo, id, GIT_OBJ_COMMIT);

00056 /**
00057  * Close an open commit
00058  *
00059  * This is a wrapper around git_object_close()
00060  *
00061  * IMPORTANT:
00062  * It *is* necessary to call this method when you stop
00063  * using a commit. Failure to do so will cause a memory leak.
00064  *
00065  * @param commit the commit to close
00066  */
00067 
00068 GIT_INLINE(void) git_commit_close(git_commit *commit)
00069 {
00070         git_object_close((git_object *) commit);
00071 }
