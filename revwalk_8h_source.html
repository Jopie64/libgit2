00041 /**
00042  * Sort the repository contents in no particular ordering;
00043  * this sorting is arbitrary, implementation-specific

00049 /**

00054 #define GIT_SORT_TOPOLOGICAL  (1 << 0)

00056 /**

00061 #define GIT_SORT_TIME         (1 << 1)

00063 /**

00070 /**
00071  * Allocate a new revision walker to iterate through a repo.
00072  *
00073  * This revision walker uses a custom memory pool and an internal
00074  * commit cache, so it is relatively expensive to allocate.
00075  *
00076  * For maximum performance, this revision walker should be
00077  * reused for different walks.
00078  *
00079  * This revision walker is *not* thread safe: it may only be
00080  * used to walk a repository on a single thread; however,
00081  * it is possible to have several revision walkers in
00082  * several different threads walking the same repository.
00083  *
00084  * @param walker pointer to the new revision walker
00085  * @param repo the repo to walk through
00086  * @return 0 on success; error code otherwise
00087  */
00088 GIT_EXTERN(int) git_revwalk_new(git_revwalk **walker, git_repository *repo);
00089 
00090 /**
00091  * Reset the revision walker for reuse.
00092  *
00093  * This will clear all the pushed and hidden commits, and
00094  * leave the walker in a blank state (just like at
00095  * creation) ready to receive new commit pushes and
00096  * start a new walk.
00097  *
00098  * The revision walk is automatically reset when a walk
00099  * is over.
00100  *
00101  * @param walker handle to reset.
00102  */
00103 GIT_EXTERN(void) git_revwalk_reset(git_revwalk *walker);

00138 /**
00139  * Get the next commit from the revision walk.
00140  *
00141  * The initial call to this method is *not* blocking when
00142  * iterating through a repo with a time-sorting mode.
00143  *
00144  * Iterating with Topological or inverted modes makes the initial

00148  *

00150  *
00151  * @param oid Pointer where to store the oid of the next commit
